**Goal:** Add cost tracking that connects receipt data to recipes, meal plans, and grocery lists — with a price database, budget reports, inline cost alerts, and store-aware pricing.

**Architecture:** Four new services (price-store, cost-estimator, budget-reporter, budget-alerts) and one new handler (budget). Price data flows from receipt photos, text messages, or manual Obsidian edits into per-store markdown price files. Cost estimation uses LLM semantic matching + unit conversion against the price database. Budget reports are generated on-demand and persisted as cost-history markdown files.

**Tech Stack:** TypeScript, Vitest, YAML, Obsidian-compatible markdown with frontmatter, LLM (standard tier) for semantic matching and unit conversion.

export interface PriceEntry {
	name: string;          // normalized item name, e.g. "Eggs (60ct)"
	price: number;         // dollar amount
	unit: string;          // package unit, e.g. "60ct", "1 gal", "5 lb"
	department: string;    // Dairy, Produce, Meat, Pantry, etc.
	updatedAt: string;     // ISO date of last update
}

export interface StorePriceData {
	store: string;         // display name, e.g. "Costco"
	slug: string;          // file-safe name, e.g. "costco"
	lastUpdated: string;   // ISO date
	items: PriceEntry[];
}

export interface IngredientCost {
	ingredientName: string;    // from recipe, e.g. "2 cups AP flour"
	matchedItem: string | null; // from price DB, e.g. "AP flour (25 lb)"
	matchedPrice: number | null; // full package price
	matchedUnit: string | null;  // package unit
	portionCost: number;        // cost for the recipe's quantity
	isEstimate: boolean;        // true if LLM-estimated (no price DB match)
}

export interface MealCostEstimate {
	recipeId: string;
	recipeTitle: string;
	store: string;            // which store's prices were used
	ingredientCosts: IngredientCost[];
	totalCost: number;        // sum of portionCost
	perServingCost: number;   // totalCost / servings
	servings: number;
	estimatedAt: string;      // ISO datetime
}

/**
 * Price store — per-store price database CRUD.
 *
 * Manages Obsidian-compatible .md files in shared/prices/.
 * Each store has its own file (e.g., prices/costco.md) with items
 * organized by department.
 */

import type { ScopedDataStore } from '@pas/core/types';
import { buildAppTags, generateFrontmatter } from '@pas/core/utils/frontmatter';
import type { PriceEntry, StorePriceData } from '../types.js';

const PRICES_DIR = 'prices';

// ─── Slug Helpers ──────────────────────────────────────────────

/** Convert a store display name to a file-safe slug. */
export function getStoreSlug(storeName: string): string {
	return storeName
		.toLowerCase()
		.replace(/[^a-z0-9\s-]/g, '')
		.replace(/\s+/g, '-')
		.replace(/-+/g, '-')
		.replace(/^-|-$/g, '');
}

// ─── File Format ───────────────────────────────────────────────

/** Format a StorePriceData as an Obsidian-compatible markdown file. */
export function formatPriceFile(data: StorePriceData): string {
	const fm = generateFrontmatter({
		store: data.store,
		slug: data.slug,
		last_updated: data.lastUpdated,
		item_count: data.items.length,
		tags: buildAppTags('food', 'prices'),
		app: 'food',
	});

	// Group items by department
	const byDept = new Map<string, PriceEntry[]>();
	for (const item of data.items) {
		const dept = item.department || 'Other';
		const existing = byDept.get(dept) ?? [];
		existing.push(item);
		byDept.set(dept, existing);
	}

	const lines: string[] = [];
	for (const [dept, items] of byDept) {
		lines.push(`## ${dept}`);
		for (const item of items) {
			lines.push(`- ${item.name}: $${item.price.toFixed(2)} <!-- updated: ${item.updatedAt} -->`);
		}
		lines.push('');
	}

	return fm + '\n' + lines.join('\n');
}

/** Line pattern: "- Item Name (unit): $12.34 <!-- updated: 2026-04-05 -->" */
const PRICE_LINE_RE = /^- (.+?):\s*\$(\d+(?:\.\d+)?)\s*(?:<!--\s*updated:\s*(\d{4}-\d{2}-\d{2})\s*-->)?$/;

/** Extract unit from parenthetical at end of name, e.g. "Eggs (60ct)" → "60ct" */
function extractUnit(name: string): string {
	const match = name.match(/\(([^)]+)\)\s*$/);
	return match?.[1] ?? '';
}

/** Parse a formatted price file back to StorePriceData. */
export function parsePriceFile(raw: string, slug: string): StorePriceData {
	if (!raw.trim()) {
		return { store: slug, slug, lastUpdated: '', items: [] };
	}

	// Parse frontmatter
	let store = slug;
	let lastUpdated = '';
	const fmMatch = raw.match(/^---\n([\s\S]*?)\n---/);
	if (fmMatch) {
		const fmBlock = fmMatch[1] ?? '';
		const storeMatch = fmBlock.match(/^store:\s*(.+)$/m);
		if (storeMatch) store = storeMatch[1]!.trim();
		const dateMatch = fmBlock.match(/^last_updated:\s*"?(\d{4}-\d{2}-\d{2})"?$/m);
		if (dateMatch) lastUpdated = dateMatch[1]!;
	}

	// Parse items by department
	const items: PriceEntry[] = [];
	let currentDept = 'Other';

	for (const line of raw.split('\n')) {
		const deptMatch = line.match(/^##\s+(.+)$/);
		if (deptMatch) {
			currentDept = deptMatch[1]!.trim();
			continue;
		}

		const priceMatch = line.match(PRICE_LINE_RE);
		if (priceMatch) {
			const name = priceMatch[1]!.trim();
			items.push({
				name,
				price: Number.parseFloat(priceMatch[2]!),
				unit: extractUnit(name),
				department: currentDept,
				updatedAt: priceMatch[3] ?? lastUpdated,
			});
		}
	}

	return { store, slug, lastUpdated, items };
}

// ─── Store Operations ──────────────────────────────────────────

/** Load price data for a specific store. Returns empty data if file doesn't exist. */
export async function loadStorePrices(store: ScopedDataStore, storeSlug: string): Promise<StorePriceData> {
	const raw = await store.read(`${PRICES_DIR}/${storeSlug}.md`);
	if (!raw) return { store: storeSlug, slug: storeSlug, lastUpdated: '', items: [] };
	return parsePriceFile(raw, storeSlug);
}

/** Save price data for a store. */
export async function saveStorePrices(store: ScopedDataStore, data: StorePriceData): Promise<void> {
	const content = formatPriceFile(data);
	await store.write(`${PRICES_DIR}/${data.slug}.md`, content);
}

/** Add or update a price entry. Matches by name (case-insensitive). */
export function addOrUpdatePrice(data: StorePriceData, entry: PriceEntry): StorePriceData {
	const lowerName = entry.name.toLowerCase();
	const idx = data.items.findIndex((i) => i.name.toLowerCase() === lowerName);

	const items = [...data.items];
	if (idx >= 0) {
		// Preserve original casing, update everything else
		items[idx] = { ...entry, name: items[idx]!.name };
	} else {
		items.push(entry);
	}

	return { ...data, items, lastUpdated: entry.updatedAt };
}

/** Look up a price by exact name match (case-insensitive). Returns null if not found. */
export function lookupPrice(items: PriceEntry[], name: string): PriceEntry | null {
	const lower = name.toLowerCase();
	return items.find((i) => i.name.toLowerCase() === lower) ?? null;
}

/** List all store slugs from the prices directory. */
export async function listStores(store: ScopedDataStore): Promise<string[]> {
	const files = await store.list(PRICES_DIR);
	return files
		.filter((f) => f.endsWith('.md'))
		.map((f) => f.replace(/\.md$/, ''));
}

- Normalize abbreviations to full words

- Use title case for item names

	addedCount: number;
	error?: string;
}

/** Update price store from a parsed receipt. Uses LLM to normalize item names. */
export async function updatePricesFromReceipt(
	services: CoreServices,
	store: ScopedDataStore,
	receipt: Receipt,
): Promise<ReceiptUpdateResult> {
	// Filter to valid priced items
	const validItems = receipt.lineItems.filter((li) => li.totalPrice > 0);
	if (validItems.length === 0) return { updatedCount: 0, addedCount: 0 };

	const slug = getStoreSlug(receipt.store);

	let normalized: NormalizedItem[];
	try {
		const itemList = validItems.map((li) => `"${sanitizeInput(li.name, 100)}"`).join(', ');
		const result = await services.llm.complete(
			`${NORMALIZE_PROMPT}\n\nReceipt items: [${itemList}]`,
			{ tier: 'fast' },
		);
		const cleaned = result.replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '').trim();
		normalized = JSON.parse(cleaned) as NormalizedItem[];
	} catch (err) {
		services.logger.error('Failed to normalize receipt items: %s', err);
		return { updatedCount: 0, addedCount: 0, error: String(err) };
	}

	// Build a map from receipt name to normalized data
	const normalMap = new Map(normalized.map((n) => [n.receiptName, n]));

	// Load existing price data
	let priceData = await loadStorePrices(store, slug);
	if (!priceData.store || priceData.store === slug) {
		priceData = { ...priceData, store: receipt.store, slug };
	}

	let updatedCount = 0;
	let addedCount = 0;

	for (const li of validItems) {
		const norm = normalMap.get(li.name);
		if (!norm) continue;

		const entry: PriceEntry = {
			name: norm.normalizedName,
			price: li.totalPrice,
			unit: norm.unit,
			department: norm.department,
			updatedAt: receipt.date,
		};

		const existing = lookupPrice(priceData.items, norm.normalizedName);
		priceData = addOrUpdatePrice(priceData, entry);
		if (existing) {
			updatedCount++;
		} else {
			addedCount++;
		}
	}

	await saveStorePrices(store, priceData);

	return { updatedCount, addedCount };
}

// ─── Text Intent Parsing ───────────────────────────────────────

const PRICE_UPDATE_RE = /\$\d+(?:\.\d+)?/;
const PRICE_UPDATE_VERBS = /\b(cost|price|are|is|was|costs|now|update|set)\b/i;

/** Detect if a message is a price update intent (contains $ amount + price-related verb). */
export function isPriceUpdateIntent(text: string): boolean {
	return PRICE_UPDATE_RE.test(text) && PRICE_UPDATE_VERBS.test(text);
}

export interface ParsedPriceUpdate {
	item: string;
	price: number;
	store: string;
	unit: string;
	department: string;
}

const PARSE_PRICE_PROMPT = `Extract a price update from this message. Return ONLY valid JSON (no markdown, no explanation):
{
  "item": "Eggs (60ct)",
  "price": 3.50,
  "store": "Costco",
  "unit": "60ct",
  "department": "Dairy"
}

Rules:
- item: normalized name with package size in parentheses
- price: the dollar amount mentioned
- store: the store mentioned, or "Unknown" if not specified
- unit: package size
- department: one of Dairy, Produce, Meat, Seafood, Bakery, Pantry, Frozen, Beverages, Snacks, Other`;

/** Parse a free-text price update message using LLM. Returns null if parsing fails. */
export async function parsePriceUpdateText(
	services: CoreServices,
	text: string,
): Promise<ParsedPriceUpdate | null> {
	try {
		const result = await services.llm.complete(
			`${PARSE_PRICE_PROMPT}\n\nMessage: "${sanitizeInput(text, 200)}"`,
			{ tier: 'fast' },
		);
		const cleaned = result.replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '').trim();
		const parsed = JSON.parse(cleaned) as ParsedPriceUpdate;
		if (!parsed.item || typeof parsed.price !== 'number') return null;
		return parsed;
	} catch (err) {
		services.logger.error('Failed to parse price update text: %s', err);
		return null;
	}
}

 * Cost estimator — LLM-powered ingredient-to-price matching with unit conversion.
 *
 * Uses semantic matching to map recipe ingredients to the price database,
 * then calculates proportional costs based on package sizes.
 */

import type { CoreServices } from '@pas/core/types';
import type {
	GroceryItem,
	IngredientCost,
	MealCostEstimate,
	MealPlan,
	PriceEntry,
	Recipe,
} from '../types.js';

import { sanitizeInput } from '../utils/sanitize.js';

// ─── LLM Prompts ──────────────────────────────────────────────

const ESTIMATE_PROMPT = `You are a grocery cost calculator. Given recipe ingredients and a price database, match each ingredient to the closest item in the database and calculate the proportional cost.

Return ONLY a JSON array (no markdown, no explanation):
[
  {
    "ingredientName": "2 cups AP flour",
    "matchedItem": "AP Flour (25 lb)",
    "portionCost": 0.20,
    "isEstimate": false
  }
]

Rules:
- Match each ingredient to the most similar item in the price database
- Calculate portionCost: what fraction of the package the recipe needs, times the price
- Use unit conversion: 1 lb = 16 oz = ~2 cups flour, 1 gal = 16 cups, etc.
- If no match in the database, set matchedItem to null, estimate a reasonable portionCost, and set isEstimate to true
- portionCost must be a positive number (the cost for just the recipe's quantity)`;

const GROCERY_COST_PROMPT = `You are a grocery cost estimator. Match each grocery list item to the closest item in the price database.

Return ONLY a JSON array (no markdown, no explanation):
[
  { "itemName": "Eggs", "matchedItem": "Eggs (60ct)", "estimatedCost": 8.49 }
]

Rules:
- Match each grocery item to the most similar price entry
- estimatedCost is the full package price (not proportional — you're buying the whole thing)
- If no match, estimate a reasonable cost and set matchedItem to null`;

// ─── Recipe Cost Estimation ────────────────────────────────────

/** Estimate cost for a single recipe using the price database. */
export async function estimateRecipeCost(
	services: CoreServices,
	recipe: Recipe,
	priceItems: PriceEntry[],
	storeName: string,
): Promise<MealCostEstimate> {
	if (recipe.ingredients.length === 0) {
		return {
			recipeId: recipe.id,
			recipeTitle: recipe.title,
			store: storeName,
			ingredientCosts: [],
			totalCost: 0,
			perServingCost: 0,
			servings: recipe.servings,
			estimatedAt: isoNow(),
		};
	}

	let ingredientCosts: IngredientCost[];
	try {
		const ingredientList = recipe.ingredients
			.map((i) => {
				const qty = i.quantity != null ? `${i.quantity}` : '';
				const unit = i.unit ?? '';
				return `${qty} ${unit} ${i.name}`.trim();
			})
			.join('\n');

		const priceList = priceItems
			.map((p) => `${p.name}: $${p.price.toFixed(2)} (${p.unit})`)
			.join('\n');

		const prompt = `${ESTIMATE_PROMPT}\n\nRecipe ingredients:\n${sanitizeInput(ingredientList, 2000)}\n\nPrice database:\n${sanitizeInput(priceList, 4000)}`;

		const result = await services.llm.complete(prompt, { tier: 'standard' });
		const cleaned = result.replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '').trim();
		const parsed = JSON.parse(cleaned) as Array<{
			ingredientName: string;
			matchedItem: string | null;
			portionCost: number;
			isEstimate: boolean;
		}>;

		ingredientCosts = parsed.map((p) => ({
			ingredientName: p.ingredientName,
			matchedItem: p.matchedItem,
			matchedPrice: p.matchedItem
				? (priceItems.find((pi) => pi.name === p.matchedItem)?.price ?? null)
				: null,
			matchedUnit: p.matchedItem
				? (priceItems.find((pi) => pi.name === p.matchedItem)?.unit ?? null)
				: null,
			portionCost: Math.max(0, p.portionCost),
			isEstimate: p.isEstimate ?? false,
		}));
	} catch (err) {
		services.logger.error('Failed to estimate recipe cost for %s: %s', recipe.title, err);
		return {
			recipeId: recipe.id,
			recipeTitle: recipe.title,
			store: storeName,
			ingredientCosts: [],
			totalCost: 0,
			perServingCost: 0,
			servings: recipe.servings,
			estimatedAt: isoNow(),
		};
	}

	const totalCost = ingredientCosts.reduce((sum, c) => sum + c.portionCost, 0);
	const perServingCost = recipe.servings > 0 ? totalCost / recipe.servings : totalCost;

	return {
		recipeId: recipe.id,
		recipeTitle: recipe.title,
		store: storeName,
		ingredientCosts,
		totalCost,
		perServingCost,
		servings: recipe.servings,
		estimatedAt: isoNow(),
	};
}

export interface GroceryListCostResult {

		name: string;
		matchedItem: string | null;
		estimatedCost: number;
	}>;
	total: number;
	store: string;

/** Estimate cost for grocery list items. */
export async function estimateGroceryListCost(
	services: CoreServices,
	groceryItems: GroceryItem[],
	priceItems: PriceEntry[],
	storeName: string,
): Promise<GroceryListCostResult> {
	if (groceryItems.length === 0) {
		return { items: [], total: 0, store: storeName };
	}

	try {
		const itemList = groceryItems
			.map((i) => {
				const qty = i.quantity != null ? `${i.quantity}` : '';
				const unit = i.unit ?? '';
				return `${qty} ${unit} ${i.name}`.trim();
			})
			.join('\n');

		const priceList = priceItems
			.map((p) => `${p.name}: $${p.price.toFixed(2)} (${p.unit})`)
			.join('\n');

		const prompt = `${GROCERY_COST_PROMPT}\n\nGrocery list:\n${sanitizeInput(itemList, 2000)}\n\nPrice database:\n${sanitizeInput(priceList, 4000)}`;
		const result = await services.llm.complete(prompt, { tier: 'standard' });
		const cleaned = result.replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '').trim();
		const parsed = JSON.parse(cleaned) as Array<{
			itemName: string;
			matchedItem: string | null;
			estimatedCost: number;
		}>;

		const items = parsed.map((p) => ({
			name: p.itemName,
			matchedItem: p.matchedItem,
			estimatedCost: Math.max(0, p.estimatedCost),
		}));

		const total = items.reduce((sum, i) => sum + i.estimatedCost, 0);

	} catch (err) {
		services.logger.error('Failed to estimate grocery list cost: %s', err);
		return { items: [], total: 0, store: storeName };
	}
}

// ─── ISO Week Helper ───────────────────────────────────────────

function isoWeek(dateStr: string): { year: number; week: string } {
	const d = new Date(dateStr + 'T00:00:00Z');
	const dayOfWeek = d.getUTCDay() || 7;
	d.setUTCDate(d.getUTCDate() + 4 - dayOfWeek);
	const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
	const weekNum = Math.ceil(((d.getTime() - yearStart.getTime()) / 86400000 + 1) / 7);
	return { year: d.getUTCFullYear(), week: String(weekNum).padStart(2, '0') };
}

function dayAbbrev(dateStr: string): string {
	const d = new Date(dateStr + 'T00:00:00Z');
	return d.toLocaleDateString('en-US', { weekday: 'short', timeZone: 'UTC' });
}

/** Generate a weekly cost report from a meal plan and its cost estimates. */
export function generateWeeklyReport(plan: MealPlan, estimates: MealCostEstimate[]): CostHistoryWeek {
	const { year, week } = isoWeek(plan.startDate);
	const weekId = `${year}-W${week}`;

	const estimateMap = new Map(estimates.map((e) => [e.recipeId, e]));

	const meals = plan.meals
		.map((m) => {
			const est = estimateMap.get(m.recipeId);
			if (!est) return null;
			return {
				date: m.date,
				recipeTitle: m.recipeTitle,
				cost: est.totalCost,
				perServing: est.perServingCost,
			};
		})
		.filter((m): m is NonNullable<typeof m> => m !== null);

	const totalCost = meals.reduce((sum, m) => sum + m.cost, 0);
	const mealCount = meals.length;
	const avgPerMeal = mealCount > 0 ? totalCost / mealCount : 0;
	const totalServingCost = meals.reduce((sum, m) => sum + m.perServing, 0);
	const avgPerServing = mealCount > 0 ? totalServingCost / mealCount : 0;

	return {
		weekId,
		startDate: plan.startDate,
		endDate: plan.endDate,
		meals,
		totalCost,
		avgPerMeal,
		avgPerServing,
		mealCount,
	};
}

/** Generate a monthly report from weekly histories. */
export function generateMonthlyReport(monthId: string, weeks: CostHistoryWeek[]): CostHistoryMonth {
	const weekSummaries = weeks.map((w) => ({
		weekId: w.weekId,
		totalCost: w.totalCost,
		mealCount: w.mealCount,
	}));

	const totalCost = weeks.reduce((sum, w) => sum + w.totalCost, 0);
	const mealCount = weeks.reduce((sum, w) => sum + w.mealCount, 0);
	const avgPerMeal = mealCount > 0 ? totalCost / mealCount : 0;
	const totalServing = weeks.reduce((sum, w) => sum + w.avgPerServing * w.mealCount, 0);
	const avgPerServing = mealCount > 0 ? totalServing / mealCount : 0;

	return { monthId, weeks: weekSummaries, totalCost, avgPerMeal, avgPerServing, mealCount };
}

/** Generate yearly summary. */
export function generateYearlyReport(
	_year: string,
	months: Array<{ monthLabel: string; totalCost: number; mealCount: number }>,
): { totalCost: number; avgPerMonth: number; months: typeof months } {
	const totalCost = months.reduce((sum, m) => sum + m.totalCost, 0);
	const avgPerMonth = months.length > 0 ? totalCost / months.length : 0;
	return { totalCost, avgPerMonth, months };
}

/** Format a weekly report for Telegram output. */
export function formatWeeklyReportMessage(week: CostHistoryWeek, prevWeek: CostHistoryWeek | null): string {
	const lines: string[] = [];

	// Header — use the start date to show "Week of Apr 7"
	const startDateFormatted = new Date(week.startDate + 'T00:00:00Z')
		.toLocaleDateString('en-US', { month: 'short', day: 'numeric', timeZone: 'UTC' });
	lines.push(`📊 Food Budget — Week of ${startDateFormatted}`);
	lines.push('');

	// Meal breakdown
	if (week.meals.length > 0) {
		lines.push('**This Week\'s Meals**');
		for (const meal of week.meals) {
			const day = dayAbbrev(meal.date);
			lines.push(`${day}: ${meal.recipeTitle} — **$${meal.cost.toFixed(2)}**`);
		}
		lines.push('');
	}

	// Totals
	lines.push(`**Weekly Total: $${week.totalCost.toFixed(2)}**`);
	lines.push(`Avg per meal: $${week.avgPerMeal.toFixed(2)} · Per person: $${week.avgPerServing.toFixed(2)}`);

	// Comparison
	if (prevWeek && prevWeek.totalCost > 0) {
		const diff = week.totalCost - prevWeek.totalCost;
		const pct = Math.abs(diff / prevWeek.totalCost * 100).toFixed(0);
		const arrow = diff < 0 ? '↓' : diff > 0 ? '↑' : '→';
		lines.push(`${arrow} ${pct}% vs last week ($${prevWeek.totalCost.toFixed(2)})`);
	}

	// Most/least expensive
	if (week.meals.length > 1) {
		const sorted = [...week.meals].sort((a, b) => b.cost - a.cost);
		const most = sorted[0]!;
		const least = sorted[sorted.length - 1]!;
		lines.push('');
		lines.push(`Most expensive: ${most.recipeTitle} ($${most.cost.toFixed(2)})`);
		lines.push(`Cheapest: ${least.recipeTitle} ($${least.cost.toFixed(2)})`);
	}

	return lines.join('\n');
}

}

/**
 * Budget alerts — flag expensive meal plans with swap suggestions.
 */

import type { CostHistoryWeek, MealCostEstimate } from '../types.js';

const ALERT_THRESHOLD_PCT = 15; // alert when >15% above rolling average

export interface BudgetAlert {
	projectedCost: number;
	averageCost: number;
	percentAbove: number;
	mostExpensiveMeal: { title: string; cost: number };
}

/**
 * Check if a set of meal cost estimates exceeds the rolling average.
 * Returns an alert object if projected cost is >15% above the average of recent weeks,
 * or null if within budget.
 */
export function checkBudgetAlert(
	estimates: MealCostEstimate[],
	recentWeeks: CostHistoryWeek[],
): BudgetAlert | null {
	if (recentWeeks.length === 0 || estimates.length === 0) return null;

	// Calculate rolling average weekly cost
	const avgWeeklyCost = recentWeeks.reduce((sum, w) => sum + w.totalCost, 0) / recentWeeks.length;
	if (avgWeeklyCost <= 0) return null;

	// Project current plan total
	const projectedCost = estimates.reduce((sum, e) => sum + e.totalCost, 0);

	const percentAbove = ((projectedCost - avgWeeklyCost) / avgWeeklyCost) * 100;

	if (percentAbove <= ALERT_THRESHOLD_PCT) return null;

	// Find most expensive meal
	const sorted = [...estimates].sort((a, b) => b.totalCost - a.totalCost);
	const most = sorted[0]!;

	return {
		projectedCost,
		averageCost: avgWeeklyCost,
		percentAbove: Math.round(percentAbove),
		mostExpensiveMeal: { title: most.recipeTitle, cost: most.totalCost },
	};
}

/** Format a budget alert for display (appended to meal plan output). */
export function formatBudgetAlert(alert: BudgetAlert): string {
	const lines: string[] = [];

/** Handle /foodbudget command. Args: [] (weekly), ["month"], ["year"]. */
export async function handleBudgetCommand(
	services: CoreServices,
	args: string[],
	userId: string,
	store: ScopedDataStore,
): Promise<void> {
	const subcommand = args[0]?.toLowerCase();

	if (subcommand === 'month') {
		await handleMonthlyReport(services, userId, store);
	} else if (subcommand === 'year') {
		await handleYearlyReport(services, userId, store);
	} else {

// ─── Weekly Report ─────────────────────────────────────────────

async function handleWeeklyReport(
	services: CoreServices,
	userId: string,
	store: ScopedDataStore,
): Promise<void> {
	// Load current plan
	const plan = await loadCurrentPlan(store);
	if (!plan) {
		await services.telegram.send(userId, '📊 No active meal plan. Create one with "plan meals for this week" to see cost tracking.');
		return;
	}

	// Load recipes and price data
	const recipes = await loadAllRecipes(store);
	const defaultStore = (await services.config.get<string>('default_store')) ?? '';
	const stores = await listStores(store);
	const storeSlug = defaultStore ? defaultStore.toLowerCase().replace(/\s+/g, '-') : stores[0] ?? '';

	let priceItems = storeSlug ? (await loadStorePrices(store, storeSlug)).items : [];

	// Estimate costs
	const estimates = priceItems.length > 0
		? await estimatePlanCost(services, plan, recipes, priceItems, storeSlug)
		: [];

	const week = generateWeeklyReport(plan, estimates);

	// Load previous week for comparison
	const allWeeks = await listWeeklyHistories(store);
	const prevWeekId = allWeeks.length > 0 ? allWeeks[allWeeks.length - 1] : null;
	const prevWeek = prevWeekId ? await loadWeeklyHistory(store, prevWeekId) : null;

	// Save current week history
	if (week.mealCount > 0) {
		await saveWeeklyHistory(store, week);
	}

	const msg = week.mealCount > 0
		? formatWeeklyReportMessage(week, prevWeek)
		: '📊 No cost data available yet. Send receipt photos or set item prices to start tracking.';

	await services.telegram.send(userId, msg);
}

// ─── Monthly Report ────────────────────────────────────────────

async function handleMonthlyReport(
	services: CoreServices,
	userId: string,
	store: ScopedDataStore,
): Promise<void> {
	const today = todayDate(services.timezone);
	const currentMonth = today.slice(0, 7); // "2026-04"
	const monthLabel = new Date(today + 'T00:00:00Z')
		.toLocaleDateString('en-US', { month: 'long', year: 'numeric', timeZone: 'UTC' });

	// Load all weekly histories for this month
	const allWeeks = await listWeeklyHistories(store);
	const monthWeeks: CostHistoryWeek[] = [];
	for (const weekId of allWeeks) {
		const week = await loadWeeklyHistory(store, weekId);
		if (week && week.startDate.startsWith(currentMonth)) {
			monthWeeks.push(week);
		}
	}

	if (monthWeeks.length === 0) {
		await services.telegram.send(userId, `📊 No cost data for ${monthLabel}. Use /foodbudget to generate a weekly report first.`);
		return;
	}

	// Load previous month total for comparison
	const prevMonth = getPrevMonth(currentMonth);
	let prevTotal: number | null = null;
	for (const weekId of allWeeks) {
		const week = await loadWeeklyHistory(store, weekId);
		if (week && week.startDate.startsWith(prevMonth)) {
			prevTotal = (prevTotal ?? 0) + week.totalCost;
		}
	}

	const msg = formatMonthlyReportMessage(monthLabel, monthWeeks, prevTotal);
	await services.telegram.send(userId, msg);
}

// ─── Yearly Report ─────────────────────────────────────────────

async function handleYearlyReport(
	services: CoreServices,
	userId: string,
	store: ScopedDataStore,
): Promise<void> {
	const today = todayDate(services.timezone);
	const currentYear = today.slice(0, 4);

	// Load all weekly histories for this year
	const allWeeks = await listWeeklyHistories(store);
	const yearWeeks: CostHistoryWeek[] = [];
	for (const weekId of allWeeks) {
		const week = await loadWeeklyHistory(store, weekId);
		if (week && week.startDate.startsWith(currentYear)) {
			yearWeeks.push(week);
		}
	}

	if (yearWeeks.length === 0) {
		await services.telegram.send(userId, `📊 No cost data for ${currentYear}. Use /foodbudget to generate a weekly report first.`);
		return;
	}

	// Aggregate by month
	const monthMap = new Map<string, { totalCost: number; mealCount: number }>();
	for (const week of yearWeeks) {
		const monthKey = week.startDate.slice(0, 7);
		const existing = monthMap.get(monthKey) ?? { totalCost: 0, mealCount: 0 };
		existing.totalCost += week.totalCost;
		existing.mealCount += week.mealCount;
		monthMap.set(monthKey, existing);
	}

	const months = [...monthMap.entries()]
		.sort(([a], [b]) => a.localeCompare(b))
		.map(([key, val]) => ({
			monthLabel: new Date(key + '-15T00:00:00Z').toLocaleDateString('en-US', { month: 'short', timeZone: 'UTC' }),
			totalCost: val.totalCost,
			mealCount: val.mealCount,
		}));

	const msg = formatYearlyReportMessage(currentYear, months);
	await services.telegram.send(userId, msg);
}

// ─── Helpers ───────────────────────────────────────────────────

function getPrevMonth(monthStr: string): string {
	const [yearStr, monthNum] = monthStr.split('-');
	const y = Number.parseInt(yearStr!, 10);
	const m = Number.parseInt(monthNum!, 10);
	if (m === 1) return `${y - 1}-12`;
	return `${y}-${String(m - 1).padStart(2, '0')}`;
}
